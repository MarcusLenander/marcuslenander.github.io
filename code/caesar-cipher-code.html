def encryptMessage(message, key):
    i = 0
    ciphertext = [0 for x in range(len(message))]
    message = list(message.lower())

    for char in message:
        if char.islower():
            message[i] = ord(char) - 97
            ciphertext[i] = chr((message[i] + key) % 26 + 97)
            i += 1
        else:
            ciphertext[i] = char
            i += 1

    return ''.join(ciphertext)

def decryptCiphertext(ciphertext, key):
    i = 0
    message = [0 for x in range(len(ciphertext))]
    ciphertext = list(ciphertext.lower())

    for char in ciphertext:
        if char.islower():
            ciphertext[i] = ord(char) - 97
            message[i] = chr((ciphertext[i] - key) % 26 + 97)
            i += 1
        else:
            message[i] = char
            i += 1

    return ''.join(message)

def main():
    message = "HELLO WORLD";
    key = 25

    ciphertext = encryptMessage(message, key)
    print("plaintext: " + message + "\nciphertext: " + ciphertext)

    plaintext = decryptCiphertext(ciphertext, key)
    print("ciphertext: " + ciphertext + "\nplaintext: " + plaintext)

if __name__ == "__main__":
    main()
